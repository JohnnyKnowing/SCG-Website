function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSheet();
    var parameter = e.parameter;
    
    var fname = parameter.fname;
    var lname = parameter.lname;
    var email = parameter.email;
    var phone = parameter.phone;
    var role = parameter.role;
    var who = parameter.who;
    var message = parameter.message;
    var timestamp = new Date();
    
    // Append rows to separate columns in the spreadsheet:
    // Timestamp, First Name, Last Name, Email, Phone, Role, Reaching Out About, Message
    sheet.appendRow([timestamp, fname, lname, email, phone, role, who, message]);
    
    var myEmail = "johnny@spangler.care";
    var subject = "New SCG Website Contact Form Submission";
    var body = "New client inquiry details:\n\n" +
               "First Name: " + fname + "\n" +
               "Last Name: " + lname + "\n" +
               "Email: " + email + "\n" +
               "Phone: " + phone + "\n" +
               "Reaching Out As: " + role + "\n" +
               "Reaching Out About: " + who + "\n\n" +
               "Message:\n" + message;
               
    MailApp.sendEmail(myEmail, subject, body);
    
    return ContentService.createTextOutput(JSON.stringify({"result": "success"}))
                         .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"result": "error", "error": error.toString()}))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
